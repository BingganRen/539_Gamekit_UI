document.querySelectorAll('[data-hp]').forEach(bar => {
  let hp = parseInt(bar.dataset.hp, 10) || 100;
  const max = parseInt(bar.dataset.hpMax, 10) || 100;
  const fill = bar.querySelector('.hp-fill');
  const value = bar.querySelector('.hp-value');
  const render = () => {
    const pct = Math.max(0, Math.min(100, (hp / max) * 100));
    fill.style.width = pct + '%';
    value.textContent = hp + ' / ' + max;
    bar.classList.toggle('mid', pct <= 50 && pct > 25);
    bar.classList.toggle('low', pct <= 25);
    bar.setAttribute('aria-valuenow', hp);
  };
  // wire up your +/- buttons to change `hp`, then render()
  render();
});

  50% { transform: scale(1.3); filter: brightness(2); }
  100% { transform: scale(1); filter: brightness(1); }
}
function addXP(wrap, amount) {
  let xp = parseInt(wrap.dataset.xp, 10);
  let lvl = parseInt(wrap.dataset.lvl, 10);
  const per = parseInt(wrap.dataset.xpPerLevel, 10);
  xp += amount;
  while (xp >= per) {
    xp -= per;
    lvl++;
    wrap.classList.add('level-up');
    setTimeout(() => wrap.classList.remove('level-up'), 800);
  }
  wrap.dataset.xp = xp;
  wrap.dataset.lvl = lvl;
  wrap.querySelector('.lvl-num').textContent = lvl;
  wrap.querySelector('.xp-points').textContent = xp + ' / ' + per + ' XP';
  wrap.querySelector('.xp-fill').style.width = (xp / per * 100) + '%';
}

  background: conic-gradient(
    from 0deg,
    rgba(0,0,0,.75) 0deg,
    rgba(0,0,0,.75) var(--cd-progress),

  );

.cd-skill.cooling {
  border-color: #5a6178; color: #5a6178; cursor: not-allowed;

document.querySelectorAll('.cd-skill').forEach(skill => {
  const duration = parseFloat(skill.dataset.cd) || 3;
  const timer = skill.querySelector('.cd-timer');
  skill.addEventListener('click', () => {
    if (skill.classList.contains('cooling')) return;
    skill.classList.add('cooling');
    const start = performance.now();
    (function tick(now) {
      const remaining = duration - (now - start) / 1000;
      if (remaining <= 0) {
        skill.classList.remove('cooling');
        skill.style.setProperty('--cd-progress', '0deg');
        timer.textContent = '';
        return;
      }
      skill.style.setProperty('--cd-progress',
        (remaining / duration * 360) + 'deg');
      timer.textContent = remaining.toFixed(1);
      requestAnimationFrame(tick);
    })(performance.now());
  });
});

function spawnToast(stack, { type, title, message, life = 3500 }) {
  const el = document.createElement('div');
  el.className = 'toast ' + type;
  el.setAttribute('role', 'status');
  el.innerHTML = `
    <div class="toast-icon">${type === 'success' ? '✓' : '⚠'}</div>
    <div class="toast-body">
      <div class="toast-title">${title}</div>
      <div class="toast-msg">${message}</div>
    </div>
    <button class="toast-close" aria-label="Dismiss">×</button>`;
  const close = () => {
    el.classList.add('leaving');
    setTimeout(() => el.remove(), 250);
  };
  el.querySelector('.toast-close').addEventListener('click', close);
  stack.appendChild(el);
  if (life > 0) setTimeout(close, life);
}
